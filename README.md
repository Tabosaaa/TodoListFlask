# TodoListFlask
A simple example of how to do a Todo list with Flask

python3 -m venv venv
. venv/bin/activate

pip install Flask
pip install Flask-SQLAlchemy

# For Mac/Linux
export FLASK_APP=app.py
export FLASK_ENV=development
flask run

# For Windows 
set FLASK_APP=app.py
set FLASK_ENV=development
flask run
